from(bucket: "drone_telemetry")
  |> range(start: -365d)
  |> filter(fn: (r) => r._measurement == "flight_hours")
  |> filter(fn: (r) => r._field == "hours" and r.drone_id == "123456789")
  |> drop(columns: ["window_start", "window_end"])   // optional, removes window tags
  |> group(columns: [])                              // combine all rows into one
  |> sum()
  |> map(fn: (r) => ({
        r with
        _field: "total_hours",
        _value: r._value
  }))
